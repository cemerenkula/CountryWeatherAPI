public void UpdateWeatherWithRequest(double latitude, double longitude)
        {

            using (WebClient web = new WebClient())
            {
                string url = $"https://api.openweathermap.org/data/2.5/weather?lat={latitude}&lon={longitude}&appid={_apiKey}&units=metric";
                var json = web.DownloadString(url);
                WeatherInfo.root Info = JsonConvert.DeserializeObject<WeatherInfo.root>(json);

                Weather weather = new Weather();
                
                Country country = _context.Countries
                    .FirstOrDefault(c => c.LatitudeRangeStart <= latitude && c.LatitudeRangeEnd >= latitude &&
                                         c.LongitudeRangeStart <= longitude && c.LongitudeRangeEnd >= longitude);
                
                weather.Latitude = (int)Info.coord.lat;
                weather.Longitude = (int)Info.coord.lon;
                weather.WeatherMain = Info.weather[0].main;
                weather.WeatherDescription = Info.weather[0].description;
                weather.Temperature = Info.main.temp;
                weather.FeelsLike = Info.main.feels_like;
                weather.Pressure = Info.main.pressure;
                weather.Humidity = Info.main.humidity;
                weather.WindSpeed = Info.wind.speed;
                weather.WindDeg = Info.wind.deg;
                weather.WindGust = Info.wind.gust;
                
                _context.Weathers.Add(weather);
                _context.Weathers.Update(weather);
                _context.SaveChanges();
                
                weather.CountryId = country.Id;
                _context.Weathers.Update(weather);
                _context.SaveChanges();
            }
        }
